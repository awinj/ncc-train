import React, {Component} from 'react';

import './index.less';

import Header from './Header';
import Left from './Left';
import Content from './Content';

class Layout extends Component {
    constructor(props) {
        super(props);

        this.getPageHeight = this.getPageHeight.bind(this);
    }

    getPageHeight() {
        let height = 'auto';
        this.page && (height = (parseInt(getComputedStyle(this.page)['height']) - 54) + 'px');

        return height;
    }

    render() {
        const {
            className,
            children
        } = this.props;
        

        return (
            <div 
                className={`layout-page-content ${className || ''}`}
                ref={ref => this.page = ref}
            >
                {React.Children.map(children, (child) => {
                    return (
                        <child.type
                            {...child.props}
                            layoutHeight={this.getPageHeight()}
                        >
                            {child.props.children}
                        </child.type>
                    );
                })}
            </div>
        );
    }
}

Layout.Content = Content;
Layout.Header = Header;
Layout.Left = Left;

export default Layout;
