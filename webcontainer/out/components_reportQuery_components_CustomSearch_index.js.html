import React, {Component} from 'react';

import './index.less';

import CustomRefer from '../CustomRefer';

import {
    base
} from 'nc-lightapp-front';

import moment from 'moment';

const {
    NCSelect,
    NCInput,
    NCDatePicker
} = base;

const {
    NCOption
} = NCSelect;

let timer = null;

class CustomSearch extends Component {
    constructor(props) {
        super(props);
    }

    componentDidMount() {
        this.hideSaveBtn();
    }

    hideSaveBtn() {
        let btn = null;
        if(document.querySelector('.NC_searchAdvModal')) {
            if(document.querySelector('.NC_searchAdvModal').querySelector('.modalFooter')) {
                if(document.querySelector('.NC_searchAdvModal').querySelector('.modalFooter').querySelector('.saveSearchPlan')) {
                    btn = document.querySelector('.NC_searchAdvModal').querySelector('.modalFooter').querySelector('.saveSearchPlan')
                }
            }
        }
        if(btn) {
            btn.innerHTML = '';
            clearTimeout(timer);
        }
        else {
            timer = setTimeout(() => {
                this.hideSaveBtn();
            }, 500);
        }
    }

    render() {
        const {
            language,
            data,
            customFormValue,
            customReferValue
        } = this.props;
        return (
            <table className="custom-search-area">
                <tr className="custom-search-row"></tr>
                <tr className="custom-search-head">
                    <td width="300">{language['i6013-000732']/*显示名*/}</td>
                    <td width="500">{language['i6013-000734']/*参数值*/}</td>
                </tr>
                {data.map((item) => {
                    let value = customFormValue[item.code];

                    if(item.refCode === '日历') {
                        value = value ? moment(value).format('YYYY-MM-DD') : null;
                    }

                    return (
                        <tr className="custom-search-row">
                            <td width="300">
                                {item.required && <span className="red-star">*</span>}
                                {item.name}
                            </td>
                            <td width="500">
                                <Choose>
                                    <When condition={/[456]/.test(item.dataType)}>
                                        <If condition={item.refCode === '日历'}>
                                            <NCDatePicker
                                                format="YYYY-MM-DD"
                                                onChange={this.props.onChange('date', item)}
                                                value={value}
                                                className="custom-search-date-picker"
                                            />
                                            <Else/>
                                            <CustomRefer
                                                refName={item.refCode}
                                                refcode={item.refPath}
                                                isMultiSelectedEnabled={item.operator === 'in'}
                                                onChange={this.props.onChange('refer', item)}
                                                value={customReferValue[item.code]}
                                                ref={ref => this.customRefer = ref}
                                            />
                                        </If>
                                    </When>
                                    <When condition={/[23]/.test(item.dataType)}>
                                        <NCSelect
                                            onChange={this.props.onChange('select', item)}
                                        >
                                            {item.options.map((item) => {
                                                return (
                                                    <NCOption
                                                        key={item}
                                                        value={item}
                                                    >
                                                        {item}
                                                    </NCOption>
                                                );
                                            })}
                                        </NCSelect>
                                    </When>
                                    <When condition={/[01]/.test(item.dataType)}>
                                        <NCInput
                                            onChange={this.props.onChange('input', item)}
                                        />
                                    </When>
                                </Choose>
                            </td>
                        </tr>
                    )
                })}
                
            </table>
        );
    }
}

export default CustomSearch;
